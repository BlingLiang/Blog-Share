## 要求

相邻的格子和对角相邻的格子中，颜色不相同，最后输出最少颜色数量

## 代码实现

```
#include<stdio.h>
/*
要求：
相邻的格子和对角相邻的格子中，颜色不相同，最后输出最少颜色数量
思路：
1.手动输入矩阵大小（容器格子数目）
2.遍历填充
  先把a[m+1][n+1]全部填充为0（方便之后的计算）
  把a[1][1]填充为1
3.然后遍历
3-1.每个有效的格子周围有8个格子，其值分别为x1...x8
    把这8个数放入1维数组b[8]
3-2.先记录下这8个格子中的最大值y
    用1~y+1遍历这8个数
    直到这8个数中，没有与i1（计数器）相同的值时，把那个i1赋值给a[i][j]
4.遍历，输出填充结果，并记录颜色数量，输出
*/

 main(){
 int i,j,k=0,i1,j1,w=0;                                     //计数器
 int a[100][100];                                     //容器
 int m,n;                                             //容器大小
 int b[8];                                            //表示周围量的数组
 int y=0;                                             // 周围数的最大值
 printf("请输入矩形规则容器大小（行数，列数）,数值均大于0\n");
 scanf("%d%d",&m,&n);
 printf("您输入的行数为：");
 printf("%d\n",m);
 printf("您输入的列数为：");
 printf("%d\n",n);

 for(i=0;i<m+2;i++){                                  //把a[m+1][n+1]全部填充为0（方便之后的计算）
    for(j=0;j<n+2;j++){
        a[i][j]=0;
    }
 }

 for(i=1;i<m+1;i++){
    for(j=1;j<n+1;j++){
        b[0]=a[i-1][j-1];                             //把这8个周围数放入1维数组b[8]
        b[1]=a[i-1][j];
        b[2]=a[i-1][j+1];
        b[3]=a[i][j-1];
        b[4]=a[i][j+1];
        b[5]=a[i+1][j-1];
        b[6]=a[i+1][j];
        b[7]=a[i+1][j+1];
        for(i1=0;i1<8;i1++){                          //先记录下这8个格子中的最大值y
           if(y<b[i1]){
            y=b[i1];
           }
        }
        for(i1=1;i1<y+2;i1++){                       //确定a[i][j]的值
            for(j1=0;j1<8;j1++){
                if(i1==b[j1]){
                    w+=1;
                }
            }
            if(w==0){
                a[i][j]=i1;
                i1=y+2;
            }
            w=0;
        }
    }
 }

 printf("填充示例：\n");
 for(i=1;i<m+1;i++){                                   //颜色数量计数
    for(j=1;j<n+1;j++){
        printf("%d\t",a[i][j]);
        if(k<a[i][j]){
            k=a[i][j];
        }
    }
    printf("\n");
 }
 printf("颜色数量：");//输出结果
 printf("%d\n",k);

 }
```
